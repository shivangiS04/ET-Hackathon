ADVANCED PYTEST OPTIONS
─────────────────────────────────────────────────────────────────────────────

# Stop on first failure
pytest -x

# Show local variables on failure
pytest -l

# Run debugger on failure (interactive)
pytest --pdb

# Verbose + detailed
pytest -vv

# Quiet output (summary only)
pytest -q

# Show slowest 10 tests
pytest --durations=10

# Run only tests matching pattern
pytest -k "battery and healthy" -v

# Run tests matching marker
pytest -m "unit" -v
pytest -m "not performance" -v

# Show all test names without running
pytest --collect-only -q

# Run with different output format
pytest --tb=short          # Short traceback
pytest --tb=line          # One line per error
pytest --tb=native        # Native Python traceback
pytest --tb=no            # No traceback
